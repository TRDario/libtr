159template <std::output_iterator<glm::vec2> It>
160void tr::fillArcVertices(It out, std::size_t vertices, CircleF circ, AngleF startth, AngleF sizeth)
161{
162 auto dth { sizeth / vertices };
163 auto dsin { dth.sin() };
164 auto dcos { dth.cos() };
165 glm::vec2 delta { circ.r * startth.cos(), circ.r * startth.sin() };
166 for (std::size_t i = 0; i < vertices; ++i) {
167 *out++ = delta + circ.c;
168 delta = glm::vec2 { dcos * delta.x - dsin * delta.y, dsin * delta.x + dcos * delta.y };
169 }
170}
