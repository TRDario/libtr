155template <std::output_iterator<glm::vec2> It>
156void tr::fillArcVertices(It out, std::size_t vertices, CircleF circ, AngleF startth, AngleF sizeth)
157{
158 auto dth{sizeth / vertices};
159 auto dsin{dth.sin()};
160 auto dcos{dth.cos()};
161 glm::vec2 delta{circ.r * startth.cos(), circ.r * startth.sin()};
162 for (std::size_t i = 0; i < vertices; ++i) {
163 *out++ = delta + circ.c;
164 delta = glm::vec2{dcos * delta.x - dsin * delta.y, dsin * delta.x + dcos * delta.y};
165 }
166}
