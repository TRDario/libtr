67template <class T1, class T2, class T3>
68constexpr bool tr::collinear(glm::tvec2<T1> a, glm::tvec2<T2> b, glm::tvec2<T3> c) noexcept
69{
70 auto tol{std::abs(std::max<std::common_type_t<T1, T2, T3>>({a.x, a.y, b.x, b.y, c.x, c.y})) * 0.000001f};
71 return std::abs((b.x - a.x) * (c.y - a.y) - (c.x - a.x) * (b.y - a.y)) < tol;
72}

99template <std::floating_point T>
100constexpr std::optional<glm::tvec2<T>> tr::segmentIntersect(glm::tvec2<T> a1, glm::tvec2<T> b1, glm::tvec2<T> a2,
101 glm::tvec2<T> b2) noexcept
102{
103 auto r{b2 - a2};
104 glm::tvec2<T> n{b1.y - a1.y, -b1.x + a1.x};
105 auto d{dot(r, n)};
106 if (!d) {
107 return std::nullopt;
108 }
109 auto t{dot(a1 - a2, n) / d};
110 if (t < 0 || t > 1) {
111 return std::nullopt;
112 }
113 auto result{a2 + r * t};
114 if ((result.x < std::min(a1.x, b1.x) || result.x > std::max(a1.x, b1.x)) ||
115 (result.x - a1.x == 0 && (result.y < std::min(a1.y, b1.y) || result.y > std::max(a1.y, b1.y)))) {
116 return std::nullopt;
117 }
118 else {
119 return result;
120 }
121}

218template <tr::Arithmetic T> T tr::mirrorRepeat(T v, T min, T max) noexcept
219{
220 auto shift{v - min};
221 auto size{max - min};
222 shift = mod(abs(shift + (shift < 0)), (2 * size));
223 return min + (shift >= size ? 2 * size - shift - 1 : shift);
224}
