608template <std::floating_point T>
609constexpr std::optional<glm::tvec2<T>>
610tr::segmentIntersect(glm::tvec2<T> a1, glm::tvec2<T> b1, glm::tvec2<T> a2, glm::tvec2<T> b2) noexcept
611{
612 auto r {b2 - a2};
613 glm::tvec2<T> n {b1.y - a1.y, -b1.x + a1.x};
614 auto d {dot(r, n)};
615 if (!d) {
616 return std::nullopt;
617 }
618 auto t {dot(a1 - a2, n) / d};
619 if (t < 0 || t > 1) {
620 return std::nullopt;
621 }
622 auto result {a2 + r * t};
623 if ((result.x < std::min(a1.x, b1.x) || result.x > std::max(a1.x, b1.x))
624 || (result.x - a1.x == 0 && (result.y < std::min(a1.y, b1.y) || result.y > std::max(a1.y, b1.y)))) {
625 return std::nullopt;
626 }
627 else {
628 return result;
629 }
630}

728template <tr::Arithmetic T>
729T tr::mirrorRepeat(T v, T min, T max) noexcept
730{
731 auto shift {v - min};
732 auto size {max - min};
733 shift = mod(abs(shift + (shift < 0)), (2 * size));
734 return min + (shift >= size ? 2 * size - shift - 1 : shift);
735}
