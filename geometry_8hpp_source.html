579template <std::floating_point T>
580constexpr std::optional<glm::tvec2<T>> tr::segmentIntersect(glm::tvec2<T> a1, glm::tvec2<T> b1, glm::tvec2<T> a2,
581 glm::tvec2<T> b2) noexcept
582{
583 auto r{b2 - a2};
584 glm::tvec2<T> n{b1.y - a1.y, -b1.x + a1.x};
585 auto d{dot(r, n)};
586 if (!d) {
587 return std::nullopt;
588 }
589 auto t{dot(a1 - a2, n) / d};
590 if (t < 0 || t > 1) {
591 return std::nullopt;
592 }
593 auto result{a2 + r * t};
594 if ((result.x < std::min(a1.x, b1.x) || result.x > std::max(a1.x, b1.x)) ||
595 (result.x - a1.x == 0 && (result.y < std::min(a1.y, b1.y) || result.y > std::max(a1.y, b1.y)))) {
596 return std::nullopt;
597 }
598 else {
599 return result;
600 }
601}

698template <tr::Arithmetic T> T tr::mirrorRepeat(T v, T min, T max) noexcept
699{
700 auto shift{v - min};
701 auto size{max - min};
702 shift = mod(abs(shift + (shift < 0)), (2 * size));
703 return min + (shift >= size ? 2 * size - shift - 1 : shift);
704}
