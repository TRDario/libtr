139auto tr::asMutObjects(std::span<std::byte, S> bytes) noexcept
140{
141 if constexpr (S != std::dynamic_extent) {
142 static_assert(S % sizeof(T) == 0, "Cannot reinterpret byte span due to size / sizeof(T) not being an integer.");
143 return std::span<T, S / sizeof(T)> {(T*)(bytes.data())};
